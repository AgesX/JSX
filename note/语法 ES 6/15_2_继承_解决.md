function Dog(a, b){
	this.a = a;
    this.b = b;
}


Dog.prototype.vv = function(){
    console.log(this.a);
};


function SuperDog(a, b, c){
	this.c = c;
  
    // 重新指定 this
    Dog.call(this, a, b);
}


//  Object.create
//  创建一个对象
//  该对象的 prototype， 指向 Dog.prototype





SuperDog.prototype = Object.create(Dog.prototype);


SuperDog.prototype.fire = function(){
    console.log(666666);
};



let ddd = new SuperDog(6, 66, 666);

ddd.vv();
